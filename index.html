import flet as ft import cv2 import numpy as np import base64 import time import sqlite3 from ultralytics import YOLO # 🧠 Kanka, yapay zeka beyni sisteme yüklendi! # --- 1. YAPAY ZEKA MODELİNİ YÜKLE --- # İlk çalıştırmada internetten küçük bir dosya indirir, sonra hep internetsiz çalışır. model = YOLO("yolov8n.pt") # --- 2. VERİTABANI MOTORU (TÜBİTAK STANDARDI) --- def db_hazirla(): conn = sqlite3.connect("cenynx_data.db") cursor = conn.cursor() cursor.execute(""" CREATE TABLE IF NOT EXISTS users ( ad TEXT PRIMARY KEY, sifre TEXT, skor INTEGER DEFAULT 0 )""") conn.commit() conn.close() def kullanıcı_kaydet_veya_guncelle(ad, sifre, skor): conn = sqlite3.connect("cenynx_data.db") cursor = conn.cursor() cursor.execute(""" INSERT INTO users (ad, sifre, skor) VALUES (?, ?, ?) ON CONFLICT(ad) DO UPDATE SET skor = ? """, (ad, sifre, skor, skor)) conn.commit() conn.close() def kullanıcı_getir(ad): conn = sqlite3.connect("cenynx_data.db") cursor = conn.cursor() cursor.execute("SELECT sifre, skor FROM users WHERE ad = ?", (ad,)) res = cursor.fetchone() conn.close() return res def liderlik_listesi_getir(): conn = sqlite3.connect("cenynx_data.db") cursor = conn.cursor() cursor.execute("SELECT ad, skor FROM users ORDER BY skor DESC LIMIT 5") veriler = cursor.fetchall() conn.close() return veriler # --- 3. ANA UYGULAMA --- def main(page: ft.Page): db_hazirla() page.title = "CENYNX AI: Yapay Zeka Geri Dönüşüm İstasyonu" page.theme_mode = "dark" page.bgcolor = "#0a0a0a" page.horizontal_alignment = "center" page.vertical_alignment = "center" user_data = {"ad": "", "sifre": "", "skor": 0} skor_yazisi = ft.Text("0", size=90, weight="bold", color="cyan") durum = ft.Text("SİSTEM YAPAY ZEKA TARAMASINA HAZIR", size=16, color="#444444", weight="bold") kamera_onizleme = ft.Image(src="", width=380, height=280) kamera_box = ft.Container(content=kamera_onizleme, border=ft.Border.all(2, "cyan"), border_radius=20, visible=False) liderlik_kolonu = ft.Column(spacing=8) # --- 📸 YAPAY ZEKA NESNE ALGILAMA MOTORU (YOLOv8) --- def kamera_ac(e): durum.value = "YAPAY ZEKA NESNEYİ GÖRÜYOR..." durum.color = "amber" kamera_box.visible = True page.update() cap = cv2.VideoCapture(0) found_type = None start_time = time.time() while True: ret, frame = cap.read() if not ret or (time.time() - start_time > 20): # Jüri rahat test etsin diye süreyi 20 saniye yaptık break # Görüntüyü YOLO yapay zekasına gönderiyoruz results = model(frame, verbose=False) for r in results: for box in r.boxes: # Güven skorunu al (% ne kadar emin?) conf = float(box.conf[0]) if conf < 0.4: # %40'tan az eminse görmezden gel, hata yapmasın continue cls = int(box.cls[0]) label = model.names[cls] # Nesnenin İngilizce adı (bottle, cup, scissors vb.) # Nesnenin koordinatlarını alıp ekrana kutu çizeceğiz kanka b = box.xyxy[0].cpu().numpy() x1, y1, x2, y2 = int(b[0]), int(b[1]), int(b[2]), int(b[3]) # --- 🧠 YAPAY ZEKA ATIK SINIFLANDIRMA MANTIĞI --- if label in ["bottle"]: found_type = ("PLASTİK", 20, "cyan") cv2.rectangle(frame, (x1, y1), (x2, y2), (255, 255, 0), 2) cv2.putText(frame, f"PLASTIK %{int(conf*100)}", (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2) elif label in ["cup", "glass"]: found_type = ("CAM", 30, "green") cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2) cv2.putText(frame, f"CAM %{int(conf*100)}", (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2) elif label in ["scissors", "fork", "knife", "spoon", "can"]: # 🦾 Metal nesneleri veya teneke kutuları doğrudan yakalar found_type = ("METAL", 40, "amber") cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 165, 255), 2) cv2.putText(frame, f"METAL %{int(conf*100)}", (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 165, 255), 2) elif label in ["book", "newspaper", "box"]: found_type = ("KAĞIT", 10, "white") cv2.rectangle(frame, (x1, y1), (x2, y2), (255, 255, 255), 2) cv2.putText(frame, f"KAGIT %{int(conf*100)}", (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 2) # Ekrana fütüristik yapay zeka analiz logu basma cv2.putText(frame, "CENYNX AI VISION MODE ACTIVE", (10, 25), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 255), 1) _, buffer = cv2.imencode(".jpg", frame) kamera_onizleme.src_base64 = base64.b64encode(buffer).decode("utf-8") page.update() if found_type: time.sleep(1.0) # Jüri ekrandaki o yapay zeka kutusunu rahatça görebilsin diye 1 saniye donuyoruz break cap.release() kamera_box.visible = False if found_type: user_data["skor"] += found_type[1] kullanıcı_kaydet_veya_guncelle(user_data["ad"], user_data["sifre"], user_data["skor"]) skor_yazisi.value = str(user_data["skor"]) durum.value = f"YAPAY ZEKA TESPİTİ BAŞARILI: {found_type[0]} (+{found_type[1]} PUAN)!" durum.color = found_type[2] liderlik_tablosunu_yenile() else: durum.value = "ZAMAN AŞIMI: NESNE AYIRT EDİLEMEDİ" durum.color = "red" page.update() # --- 📊 LİDERLİK TABLOSU --- def liderlik_tablosunu_yenile(): liderlik_kolonu.controls.clear() liderlik_kolonu.controls.append(ft.Text("📊 TÜBİTAK LİDERLİK TABLOSU (TOP 5)", size=14, weight="bold", color="grey")) skorlar = liderlik_listesi_getir() if not skorlar: liderlik_kolonu.controls.append(ft.Text("Henüz veri girişi yapılmadı.", color="white")) else: for idx, (ad, skor) in enumerate(skorlar, 1): renk = "cyan" if idx == 1 else "white" liderlik_kolonu.controls.append( ft.Row([ ft.Text(f"{idx}. {ad.upper()}", color=renk, weight="bold" if idx==1 else "normal"), ft.Text(f"{skor} PTS", color=renk) ], alignment=ft.MainAxisAlignment.SPACE_BETWEEN) ) # --- 🏗️ ANA PANEL --- def ana_paneli_ac(): page.clean() page.vertical_alignment = "start" skor_yazisi.value = str(user_data["skor"]) liderlik_tablosu = ft.Container( content=liderlik_kolonu, padding=20, bgcolor="#111111", border_radius=15, width=350 ) liderlik_tablosunu_yenile() page.add( ft.Container(height=10), ft.Text(f"AKTİF OPERATÖR: {user_data['ad'].upper()}", size=12, color="cyan", weight="bold"), skor_yazisi, durum, ft.Container(height=10), kamera_box, ft.Container(height=10), ft.ElevatedButton("♻️ YAPAY ZEKA TARAMASINI BAŞLAT", on_click=kamera_ac, width=350, height=70, style=ft.ButtonStyle(shape=ft.RoundedRectangleBorder(radius=12), bgcolor="cyan", color="black")), ft.Divider(height=30, color="transparent"), liderlik_tablosu, ft.Container(height=20), ft.Text("Developed by Çınar 'Codebreaker' Çinko", size=11, italic=True, color="#555555"), ft.Text("TÜBİTAK Nesne Algılama ve Derin Öğrenme Prototipi", size=9, color="#333333") ) page.update() # --- 🔑 GİRİŞ EKRANI --- def giris_ekrani_ac(e=None): page.clean() ad_grid = ft.TextField(label="Operatör Kimliği / Kullanıcı Adı", width=300, border_color="cyan") sifre_gir = ft.TextField(label="Sistem Şifresi", password=True, width=300, border_color="cyan") hata = ft.Text("", color="red") def kontrol(e): db_user = kullanıcı_getir(ad_grid.value) if db_user and db_user[0] == sifre_gir.value: user_data["ad"] = ad_grid.value user_data["sifre"] = sifre_gir.value user_data["skor"] = db_user[1] ana_paneli_ac() else: hata.value = "Kimlik doğrulanamadı veya şifre hatalı!" page.update() page.add( ft.Text("🔒 SİSTEME ERİŞİM", size=26, weight="bold", color="cyan"), ft.Container(height=10), ad_grid, sifre_gir, hata, ft.Container(height=10), ft.ElevatedButton("GİRİŞ YAP", on_click=kontrol, width=250, height=50, bgcolor="cyan", color="black"), ft.TextButton("Yeni Operatör Kaydı Oluştur", on_click=lambda _: kayit_ekrani_ac()) ) page.update() # --- 📝 KAYIT EKRANI --- def kayit_ekrani_ac(): page.clean() yeni_ad = ft.TextField(label="Yeni Kullanıcı Adı Belirleyin", width=300, border_color="green") yeni_sifre = ft.TextField(label="Yeni Şifre Belirleyin", password=True, width=300, border_color="green") kayit_hata = ft.Text("", color="orange") def sifre_kaydet(e): if len(yeni_ad.value) > 2 and len(yeni_sifre.value) > 2: kullanıcı_kaydet_veya_guncelle(yeni_ad.value, yeni_sifre.value, 0) giris_ekrani_ac() else: kayit_hata.value = "Kimlik ve şifre en az 3 karakter olmalı!" page.update() page.add( ft.Text("📝 YENİ OPERATÖR", size=26, weight="bold", color="green"), ft.Text("Yapay zeka istasyonu için profil oluşturun", color="grey", size=12), ft.Container(height=10), yeni_ad, yeni_sifre, kayit_hata, ft.Container(height=10), ft.ElevatedButton("PROFİLİ AKTİF ET", on_click=sifre_kaydet, width=300, height=60, bgcolor="green", color="white"), ft.TextButton("Giriş Ekranına Dön", on_click=giris_ekrani_ac) ) page.update() giris_ekrani_ac() ft.app(target=main)
